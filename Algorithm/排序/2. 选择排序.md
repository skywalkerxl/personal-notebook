``` javascript
function SelectionSort(arr){
    for (var i = 1; i < arr.length; i++) {
　　　　var key = arr[i];
　　　　var j = i - 1;
　　　　while ( arr[j] > key) {
　　　　　　arr[j + 1] = arr[j];
　　　　　    j--;
　　　　}
　　　　arr[j + 1] = key;
　　}
    return arr;
}
```
