function SelectionSort(arr){
    // 这里也是分成两层遍历，第一层遍历是从 数组的 1 位，也就是第二个数字开始，一直到数组最末尾结束
    for (var i = 1; i < arr.length; i++) {
        var key = arr[i];
        var j = i - 1;  
       
        // 第二层遍历中，主要判断 key 值(也就是)
        while ( arr[j] > key) {
　　　　　　arr[j + 1] = arr[j];
　　　　　  j--;
        }
　　　　 arr[j + 1] = key;
　　}
    return arr;
}
